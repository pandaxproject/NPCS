prontera,164,169,5	script	Corrida Maluca	715,{
	mes "["+StrNPCInfo(1)+"]";
	mes "Olá, me diga, o que quer?";
	next;
	switch (select ("Entrar"+(GetGMLevel() > 79 ? ":Configurar Prêmios":""))) {
		mes "["+StrNPCInfo(1)+"]";

		Case 1:
			if (!$CMisopen) CallSub MesClose, "A Corrida ainda não começou!";
			else if (CountItem(12016) > 0) CallSub MesClose, "Você não pode entrar com Poção do Vento!";
			else if (CountItem(662) > 0) CallSub MesClose, "Você não pode entrar com Emblema Oficial!";
			else if (CountItem(12028) > 0) CallSub MesClose, "Você não pode entrar com Caixa do Trovão!";
			else if (CheckRiding()) CallSub MesClose, "Você não pode entrar com PecoPeco!";
			else if (CheckCart()) CallSub MesClose, "Você não pode entrar com Carrinho!";
			Warp "turbo_n_1", 57, 364; End;

		Case 2:
			Mes "Prêmios Atuais:";
			For (Set .@a, 0; .@a < 3; Set .@a, .@a + 1) Mes (.@a+1)+"º -> "+GetItemName(Getd("$CMip"+(.@a+1)));
			Next;
			select("Alterar 1º:Alterar 2º:Alterar 3º");
			Mes "["+StrNPCInfo(1)+"]";
			Mes "Digite o ID:";
			Input .@ID;
			Next;
			Mes "["+StrNPCInfo(1)+"]";
			Mes "Nome: "+GetItemName(.@ID);
			If (Select ("Confirmar:Cancelar") == 2) Close;
			Next;
			Mes "["+StrNPCInfo(1)+"]";
			Mes "Mudança realizada!";
			Close2; Setd ("$CMip"+@menu, .@ID); End;

	}
	
	MesClose:
		mes getarg(0);
		close;

	OnClock1716:
		Set $CMisopen, 1;
		Announce "[Corrida Maluca]: A corrida está prestes a começar, vocês só tem 5 minutos!", bc_all, "0xFFFFFF";
		End;

	OnClock1721:
		Set $CMisopen, 0;
		Announce "[Corrida Maluca]: A corrida começou!", bc_all, "0xFFFFFF";
		End;
}
